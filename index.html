3D Carousel Slider Without Doing the Math

Carousel Resizing

t6D3.carouselRef1 = $('.carousel.carouselDim1');
 
t6D3.carouselRef1DemoSet = {
	// carousel parent
	carouselDimensionSet: true,
	carouselWidth: "50%",
	carouselHeight: "50vh"
};
 
t6D3.carouselRef1.threeSixtyDim(t6D3.carouselRef1DemoSet);
